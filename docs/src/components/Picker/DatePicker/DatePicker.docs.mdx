## API

```jsx highlight-line="2"
import React, { useState } from 'react';
import DateFnsUtils from '@date-io/date-fns'; // choose your lib
import {
  DatePicker,
  PickersUtilProvider,
  TextField,
} from '@naturacosmeticos/natds-web';

export const DatePickerWrapper = () => {
  const [selectedDate, handleDateChange] = useState(new Date());

  const textField = (textFieldProps: any) => (
    <TextField {...textFieldProps} id="random-prop-id" type="text" />
  );

  return (
    <PickersUtilProvider utils={DateFnsUtils}>
      <DatePicker
        value={selectedDate}
        TextFieldComponent={textField}
        onChange={handleDateChange}
        variant="dialog"
      />
    </PickersUtilProvider>
  );
};
```

For more advanced usages, check [Material UI Date Picker docs](https://material-ui-pickers.dev/api/DatePicker)
