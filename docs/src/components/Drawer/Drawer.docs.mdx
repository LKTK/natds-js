## Drawer API

```jsx highlight-line="2"
import { Drawer } from '@naturacosmeticos/natds-web';

<Drawer variant="temporary" open={false} anchor="left">
  {children}
</Drawer>;
```

For more advanced usages, check [Material UI Drawer docs](https://material-ui.com/components/drawer)

---

---

## Props

| Name               |                    Type                     |                             Default                              |                                                               Description                                                               |
| :----------------- | :-----------------------------------------: | :--------------------------------------------------------------: | :-------------------------------------------------------------------------------------------------------------------------------------: |
| anchor             |   'left' \| 'top' \| 'right' \| 'bottom'    |                              'left'                              |                                                 Side from which the drawer will appear.                                                 |
| children           |                    node                     |                                                                  |                                                       The contents of the drawer.                                                       |
| classes            |                   object                    |                                                                  |   Override or extend the styles applied to the component. See the [CSS API](https://material-ui.com/api/link/#css) for more details.    |
| elevation          |                   number                    |                                16                                |                                                      The elevation of the drawer.                                                       |
| ModalProps         |                   object                    |                                {}                                |                                Props applied to the [Modal](https://material-ui.com/api/modal/) element.                                |
| onClose            |                    func                     |                                                                  |                      Callback fired when the component requests to be closed. Signature: `(event: object) => void`                      |
| open               |                    bool                     |                              false                               |                                                      If true, the drawer is open.                                                       |
| PaperProps         |                   object                    |                                {}                                |                                Props applied to the [Paper](https://material-ui.com/api/paper/) element.                                |
| SlideProps         |                   object                    |                                                                  |                                Props applied to the [Slide](https://material-ui.com/api/slide/) element.                                |
| transitionDuration | number \| { enter?: number, exit?: number } | { enter: duration.enteringScreen, exit: duration.leavingScreen } | The duration for the transition, in milliseconds. You may specify a single timeout for all transitions, or individually with an object. |
| variant            | 'permanent' \| 'persistent' \| 'temporary'  |                           'temporary'                            |                                                           The variant to use.                                                           |

## DrawerHeader API

```jsx highlight-line="2"
import { DrawerHeader } from '@naturacosmeticos/natds-web';

<DrawerHeader
  primary="Title"
  secondary="Subtitle"
  avatarSrc="path/image.png"
/>;
```

---

---

## Props

| Name           |  Type  | Default |               Description                |
| :------------- | :----: | :-----: | :--------------------------------------: |
| children       |  node  |         | The custom contents of the DrawerHeader. |
| component      | string |  'div'  |        The container HTML element        |
| primary        | string |         |          The text of the title           |
| secondary      | string |         |         The text of the subtitle         |
| avatarSrc      | string |         |      The image source of the avatar      |
| avatarChildren | string |         |          The text of the avatar          |

## DrawerMenu API

```jsx highlight-line="2"
import { DrawerMenu } from '@naturacosmeticos/natds-web';

const list = [
  {
    section: 'section',
  },
  {
    name: 'Menu Item',
    icon: 'outlined-action-love',
    badge: 'Novo',
    onSelect: () => {},
  },
  {
    name: 'Menu Item',
    icon: 'outlined-action-love',
    opened: false,
    onToggle: () => {},
    badge: 'Novo',
    list: [
      {
        name: 'Menu Item',
        icon: 'outlined-action-love',
        badge: 'Novo',
      },
    ],
  },
];

<DrawerMenu list={list} />;
```

---

---

## Props

| Name      |            Type             | Default |              Description               |
| :-------- | :-------------------------: | :-----: | :------------------------------------: |
| children  |            node             |         | The custom contents of the DrawerMenu. |
| component |           string            |  'div'  |       The container HTML element       |
| list      | array[IDrawerMenuItemProps] |         |           The list of Items            |

```jsx highlight-line="2"
interface IDrawerMenuItemProps {
  badge?: string;
  icon?: keyof typeof iconNames;
  list?: [IDrawerMenuSectionProps];
  name?: string;
  onSelect?: (event: React.SyntheticEvent, name: string) => void;
  onToggle?: (event: React.SyntheticEvent, state: boolean, name: string) => void;
  opened?: boolean;
  section?: string;
  selected?: boolean;
}
```
