## API

```jsx highlight-line="2"
import { Select, FormControl, InputLabel, MenuItem, FormHelperText } from '@naturacosmeticos/natds-web';

export const SelectWrapper = () => {
  const classes = useStyles();
  const [age, setAge] = React.useState('');

  const handleChange = (event: React.ChangeEvent<{ value: unknown }>) => {
    setAge(event.target.value as string);
  };

  return (
    <FormControl
      className={classes.formControl}
      disabled={boolean('disabled', false)}
    >
      <InputLabel shrink>Teste</InputLabel>
      <Select
        value={age}
        onChange={handleChange}
        displayEmpty={boolean('displayEmpty', true)}
        disabled={boolean('disabled', false)}
        autoWidth={boolean('autoWidth', false)}
        className={classes.selectEmpty}
      >
        <MenuItem value="">
          <em>None</em>
        </MenuItem>
        <MenuItem value={10}>Ten</MenuItem>
        <MenuItem value={20}>Twenty</MenuItem>
        <MenuItem value={30}>Thirty</MenuItem>
      </Select>
      <FormHelperText>Some important helper text</FormHelperText>
    </FormControl>
  );
};

```

For more advanced usages, check [Material UI Select docs](https://material-ui.com/pt/components/selects/)
